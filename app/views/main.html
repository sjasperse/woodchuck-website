public static void Main(string[] args)
{
    // how this dependency is constructed/injected will vary by project
    ILoggingContextProvider contextProvider = new AppLoggingContextProvider();
    ILogger logger = new AggregateLogger(
            contextProvider,
            new AuditOnlyLogger(contextProvider),
            new Log4NetLogger(contextProvider));

    logger.Info("Test message", new { username = "testusername" });

    logger.Audit("Test audit message");
}
